import java.awt.event.ActionEvent; 
import java.awt.event.ActionListener; 

import javax.swing.JButton; 
import javax.swing.JFrame; 


public class InvestmentViewer1 extends JFrame { 
	//Constants to define the width and height of the JFrame 
	private static final int FRAME_WIDTH = 400; 
	private static final int FRAME_HEIGHT = 200; 
	
	//The interest rate constant for the bank account 
	private static final double INTEREST_RATE = 10; 
	
	//The initial balance that the bank account will be initialized to 
	private static final double INITIAL_BALANCE = 1000; 
	
	//The button that will be on the screen 
	private JButton button; 
	
	//The bank account 
	private BankAccount account; 
	
	public InvestmentViewer1(){
		//initialize the class variables 
		button = new JButton("Add Interest"); 
		account = new BankAccount(INITIAL_BALANCE); 
		
		//create a new instance of the ButtonActionListener class 
		ButtonActionListener listener = new ButtonActionListener(); 
		
		//add the ActionListener object to the button 
		button.addActionListener(listener); 
		
		//set the size, title, and closing operation of the JFrame 
		setSize(FRAME_WIDTH, FRAME_HEIGHT); 
		setTitle("JFrame Example"); 
		setDefaultCloseOperation(EXIT_ON_CLOSE); 
		
		//add the button to the JFrame 
		add(button); 
	} 
	
	private class ButtonActionListener implements ActionListener { 
		@Override 
		public void actionPerformed(ActionEvent arg0){ 
			//get the amount of interest that needs to be added to the account 
			double interest = account.getBalance() * INTEREST_RATE / 100; 
			
			//deposit the interest 
			account.deposit(interest); 
			
			//print out the new balance 
			System.out.println("Balance:  " + account.getBalance()); 
		} 
	} 
	
	public static void main(String[] args){ 
		//create a new instance of the InvestmentViewer1 class 
		InvestmentViewer1 gui = new InvestmentViewer1(); 
		
		//make the JFrame visible 
		gui.setVisible(true); 
	} 
}
